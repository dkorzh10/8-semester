\textbf{1. Кинетика газа}
\newline
Функция распределения $f(t, \bm{v},\bm{r})$:
 $$
\int f(t,\bm{v},\bm{r})d\bm{v}=n(t,\bm{r})\equiv n, \;\;\; \int\bm{v}f(t,\bm{v},\bm{r})d\bm{v}= n\langle \bm{v}\rangle=n\bm{V}=\bm{j}.
 $$
Среднее любой одночастичной величины $a(t,\bm{v},\bm{r})$:
 $$
\langle a\rangle= \frac{\int
a(t,\bm{v},\bm{r})f(t,\bm{v},\bm{r})d\bm{v}}{f(t,\bm{v},\bm{r})d\bm{v}}.
 $$
Тепловая скорость $\bm{u}=\bm{v}-\langle\bm{v}\rangle =\bm{v}-\bm{V}$.
\\
Плотность внутренней энергии: $\epsilon =nm\langle u^2/2\rangle$.
\\
Плотность потока тепла: $\bm{Q}=nm\langle u^2\bm{u}/2\rangle$.
\\
Тензор плотности потока импульса:  $\Pi_{ik}=nm\langle u_iu_k\rangle$.
\\
Кинетическое уравнение Больцмана:
   $$
\hat{L}f=\frac{\partial f}{\partial t}+ v_i\frac{\partial f}{\partial
x_i}+ \frac{F_i}{m}\frac{f}{\partial v_i}=\hat{I}_{st}f,
  $$
 $$
\hat{I}_{st}=\iiint W(1,2;1’,2’)\bigl[f(1’)f(2’)-f(1)f(2)\bigr]
d\bm{v}_2d\bm{v}’_1d\bm{v}’_2\, ;
 $$
 $$
\frac{\partial\langle na\rangle}{\partial t}+\frac{\partial\langle
nv_ia\rangle}{\partial x_i}=n\langle\hat{L}\rangle+\int a\hat{I}_{st}f\,
d\bm{v}.
 $$
Субстанциональная производная:
  $$
\frac{d}{dt}=\frac{\partial}{\partial t}+\bm{V}\cdot\nabla =
\frac{\partial}{\partial t}+V_i\frac{\partial}{\partial x_i}\, .
 $$
Гидродинамическая форма законов сохранения:
  $$
\frac{dn}{dt}=-\text{div}\,\bm{V},
 $$
  $$
\frac{dV_i}{dt}=-\frac{1}{mn} \frac{\partial \Pi _{ik}}{\partial
x_k}+\frac{F_i}{m}\, ,
 $$
 $$
\frac{d}{dt}\frac{mnu^2}{2}=-\text{div}\,\bm{Q}-\Pi _{ik}\frac{\partial
V_i}{\partial x_k}\, .
 $$
Локально-равновесная функция распределения:
 $$
f_0=n\left(\frac{m}{2\pi T}\right)^{3/2}\exp\left[-\frac{m(\bm{v}-\bm{V})^2}{2T}\right],
 $$
$$
\frac{n\langle u^2\rangle_0}{2} =\frac{3}{2}T,\;\;\; \Pi_{ik}^0=nT\delta_{ik}\, .
$$
Линеаризованное уравнение Больцмана: $\; f=f_0(1+\chi )$,
 $$
\hat{L}\ln f_0=\iint \left[\chi (1’)+\chi (2’)-\chi (1)-\chi (2)\right]
f_0(2)|\bm{v}_{12}|d\sigma d\bm{v}_2\, , $$
 $$
\hat{L}\ln f_0= \frac{m}{T}\left[u_iu_k-\delta _{ik}
\frac{u^2}{3}\right]\frac{\partial V_i}{\partial x_k}+
\left[\frac{mu^2}{2}-\frac{5}{2}\right]\frac{u_i}{T} \frac{\partial
T}{\partial x_i}\, .
